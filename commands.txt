//Test Connection
ansible all -i hosts.ini -m ping
//Run the Script	
ansible all -i hosts.ini -m script -a "./setup_vm.sh"
//Copy a File Only	
ansible all -i hosts.ini -m copy -a "src=./setup_vm.sh dest=/tmp/setup_vm.sh"
//Check Disk Space
ansible all -i hosts.ini -a "df -h"
//Install a Package
ansible all -i hosts.ini -m apt -a "name=vim state=present" --become
//Check Connectivity: Verify both VMs are reachable.
ansible all -i hosts.ini -m ping
//Check Uptime: See how long servers have been running.
ansible all -i hosts.ini -a "uptime"
//Check Disk Usage: View human-readable storage details.
ansible all -i hosts.ini -a "df -h"
//Check Memory Usage: See available RAM on both nodes.
ansible all -i hosts.ini -a "free -m"
//Gather All Facts: Get a detailed JSON report of system specs (IP, OS, CPU).
ansible all -i hosts.ini -m setup
